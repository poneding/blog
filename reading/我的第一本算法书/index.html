    1.1 什么是算法 #

    算法是计算机计算和解决问题的步骤。

    选择排序：

    func selectionSort(nums []int) []int {
    for i := 0; i < len(nums); i++ {
        min := i
        for j := i + 1; j < len(nums); j++ {
            if nums[j] < nums[min] {
                min = j
            }
        }
        nums[i], nums[min] = nums[min], nums[i]
    }
    return nums
}
